  <script>

    function formatCurrency(value) {

      return value.toLocaleString("en-US", { style: "currency", currency: "USD" });

    }

    function calculateMortgage() {

      const homePrice = parseFloat(document.getElementById("homePrice").value);

      const downPayment = parseFloat(document.getElementById("downPayment").value);

      const annualRate = parseFloat(document.getElementById("interestRate").value);

      const loanYears = parseInt(document.getElementById("loanYears").value);

      const messageArea = document.getElementById("messageArea");

      messageArea.innerHTML = "";


      if (isNaN(homePrice) || isNaN(downPayment) || isNaN(annualRate) || isNaN(loanYears)) {

        showError("Please enter valid numeric values in all fields."); return;

      }

      if (homePrice <= 0) { showError("Home price must be greater than zero."); return; }

      if (downPayment < 0) { showError("Down payment cannot be negative."); return; }

      if (downPayment >= homePrice) { showError("Down payment must be less than home price."); return; }

      if (annualRate < 0) { showError("Interest rate cannot be negative."); return; }

      if (loanYears <= 0) { showError("Loan term must be greater than zero."); return; }


      const loanAmount = homePrice - downPayment;

      const monthlyRate = annualRate / 100 / 12;

      const numberOfPayments = loanYears * 12;

      let monthlyPayment;


      if (monthlyRate === 0) {

        monthlyPayment = loanAmount / numberOfPayments;

      } else {

        monthlyPayment = loanAmount * (monthlyRate * Math.pow(1 + monthlyRate, numberOfPayments)) / (Math.pow(1 + monthlyRate, numberOfPayments) - 1);

      }


      const totalPayment = monthlyPayment * numberOfPayments;

      const totalInterest = totalPayment - loanAmount;


      messageArea.innerHTML = `

        <div class="result">

          <div class="result-line"><strong>Loan Amount:</strong> ${formatCurrency(loanAmount)}</div>

          <div class="result-line"><strong>Monthly Payment:</strong> ${formatCurrency(monthlyPayment)}</div>

          <div class="result-line"><strong>Total Payment:</strong> ${formatCurrency(totalPayment)}</div>

          <div class="result-line"><strong>Total Interest:</strong> ${formatCurrency(totalInterest)}</div>

        </div>

      `;

    }

    function showError(message) {

      document.getElementById("messageArea").innerHTML = `<div class="error">${message}</div>`;

    }

    function clearFields() {

      document.getElementById("homePrice").value = "";

      document.getElementById("downPayment").value = "";

      document.getElementById("interestRate").value = "";

      document.getElementById("loanYears").value = "";

      document.getElementById("messageArea").innerHTML = "";

    }

  </script>
